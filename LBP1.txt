import java.io.*;
import java.util.*;

public class Solution
{
    public static void main(String[] args) 
    {
        Scanner sc = new Scanner(System.in);
        int num = sc.nextInt();
        if(num<0)
            System.out.print("invalid");
        else if(num%2==0)
            System.out.print("even");
        else
            System.out.print("odd");
    }
}
